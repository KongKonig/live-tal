    <script>

        async function fetchNumbers() {

            try {

                let response = await fetch('numbers.txt');

                let text = await response.text();

                document.getElementById('numbers').innerText = text;

            } catch (error) {

                document.getElementById('numbers').innerText = "Kunne ikke hente tal";

            }

        }

        

        fetchNumbers();

        setInterval(fetchNumbers, 5000); // Opdaterer hver 5. sekund

    </script>
